        const grid = document.getElementById('character-grid');
        const modal = document.getElementById('modal');
        const mobileMenu = document.getElementById('mobile-menu');

        function toggleMobileMenu() {
            mobileMenu.classList.toggle('open');
            document.body.style.overflow = mobileMenu.classList.contains('open') ? 'hidden' : 'auto';
        }

        function getMediaHtml(source) {
            const isUrl = source.startsWith('http') || source.startsWith('./');
            if (isUrl) {
                return `<img src="${source}" class="keychain-img" alt="Keychain" onerror="this.parentElement.innerHTML='‚ú®'">`;
            }
            return source;
        }

        function renderCards() {
            grid.innerHTML = KEYCHAINS.map(item => `
                <div class="character-card bg-white rounded-[2.5rem] p-6 shadow-sm border border-gray-100 flex flex-col items-center text-center" 
                     onclick="openModal(${item.id})"
                     style="background: linear-gradient(180deg, white 0%, ${item.color}15 100%)">
                    
                    <div class="w-28 h-28 md:w-32 md:h-32 rounded-3xl flex items-center justify-center text-5xl md:text-6xl mb-6 shadow-inner border-4 border-white overflow-hidden floating" 
                         style="background-color: ${item.color}44">
                        ${getMediaHtml(item.image)}
                    </div>
                    
                    <h3 class="text-2xl font-bold mb-1">${item.name}</h3>
                    <p class="text-xs text-gray-400 font-bold mb-4 uppercase tracking-widest">${item.tagline}</p>
                    
                    <div class="flex flex-wrap justify-center gap-2 mt-auto">
                        ${item.traits.map(t => `<span class="trait-badge px-3 py-1 rounded-full text-[10px] font-bold text-gray-500 uppercase">${t}</span>`).join('')}
                    </div>
                </div>
            `).join('');
        }

        function openModal(id) {
            const char = KEYCHAINS.find(c => c.id === id);
            if (!char) return;

            document.getElementById('modal-color-bar').style.backgroundColor = char.color;
            const imgContainer = document.getElementById('modal-image-container');
            imgContainer.style.backgroundColor = char.color + '44';
            imgContainer.innerHTML = getMediaHtml(char.image);

            document.getElementById('modal-name').innerText = char.name;
            document.getElementById('modal-tagline').innerText = char.tagline;
            document.getElementById('modal-personality').innerText = char.personality;
            document.getElementById('modal-food').innerText = char.favFood;
            document.getElementById('modal-hobby').innerText = char.hobby;
            document.getElementById('modal-adopt-btn').href = char.adoptUrl || "#";

            modal.classList.remove('hidden');
            document.body.style.overflow = 'hidden'; 
        }

        function closeModal() {
            modal.classList.add('hidden');
            document.body.style.overflow = 'auto';
        }

        window.onclick = (e) => { if (e.target == modal) closeModal(); }
        window.onload = renderCards;
